% make some journal names shorter
\DeclareSourcemap{%
  \maps[datatype=bibtex]{%
    % Journal abbreviations
    \map{%
      \step[fieldsource=journal, match={Journal of Semantics},
        replace={J. Semant.}]%
      \step[fieldsource=journal, match={Natural Language Semantics},
        replace={Nat. Lang. Semant.}]%
      \step[fieldsource=booktitle, match={Grazer Philosophische Studien},
        replace={Grazer Philos. Stud.}]%
      \step[fieldsource=publisher, match={Cambridge University Press},
        replace={CUP}]%
      \step[fieldset=address, null] % remove address from refs
    }
  }
}
